import java.util.Arrays;
import java.util.Scanner;

public class Demo {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        char[] firstWord = scanner.nextLine().replaceAll(" ","").toCharArray();
        char[] secondWord = scanner.nextLine().replaceAll(" ","").toCharArray();

        boolean isFirst = false;

        for (int a = 0; a < Math.min(firstWord.length,secondWord.length); a++){
            if (firstWord[a]<secondWord[a]){
                isFirst=true;
                break;
            } else if (firstWord[a]>secondWord[a]) {
                isFirst=false;
                break;
            }

            if (a == Math.min(firstWord.length,secondWord.length)-1){
                if(firstWord.length<secondWord.length){
                    isFirst=true;
                } else {
                    isFirst=false;
                }
            }

        }
        String copyFirst = String.valueOf(firstWord);
        String copySecond = String.valueOf(secondWord);
        if (isFirst){
            System.out.println(copyFirst);
            System.out.println(copySecond);
        } else {
            System.out.println(copySecond);
            System.out.println(copyFirst);

        }
    }
}
